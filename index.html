// ── MSAL CONFIG ──────────────────────────────────────────────

// WICHTIG: Ersetze CLIENT_ID_HIER mit deiner eigenen Client-ID!

const MSAL_CONFIG = {

  auth: {

    clientId: 34cbcc59-807f-4a08-bc6f-4a83d209d63a,

    authority: 'https://login.microsoftonline.com/91020d46-40a5-45b1-97ee-0245a44da351',

    redirectUri: window.location.origin + window.location.pathname

  },

  cache: { cacheLocation: 'localStorage', storeAuthStateInCookie: true }

};

const SCOPES = ['Files.ReadWrite', 'User.Read'];

const FILE_NAME = 'arbeitszeit_daten.json';

const msalApp = new msal.PublicClientApplication(MSAL_CONFIG);


// ── STATE ────────────────────────────────────────────────────

const COLORS = ['#2d6a4f','#1a6b9e','#8b3a6b','#b5451b','#6b5ea8','#a07c20','#3a7d6b','#666660'];

let selColor = COLORS[0], wOff = 0, mOff = 0, isSyncing = false;

let S = {

  areas: [

    {id:'a1', name:'Unterricht', color:'#2d6a4f', target:20},

    {id:'a2', name:'Meetings', color:'#1a6b9e', target:5},

    {id:'a3', name:'Administration', color:'#8b3a6b', target:3}

  ],

  entries: []

};

let accessToken = null;


// ── AUTH ─────────────────────────────────────────────────────

async function signIn() {

  try {

    const resp = await msalApp.loginPopup({ scopes: SCOPES });

    await afterLogin(resp.account);

  } catch(e) {

    toast('Anmeldung fehlgeschlagen: ' + e.message);

  }

}


async function afterLogin(account) {

  msalApp.setActiveAccount(account);

  const initials = account.name ? account.name.split(' ').map(n=>n[0]).join('').slice(0,2).toUpperCase() : '?';

  document.getElementById('user-dot').textContent = initials;

  document.getElementById('login-screen').style.display = 'none';

  document.getElementById('app-screen').style.display = 'block';

  await getToken();

  await loadFromOneDrive();

  init();

}


async function getToken() {

  const account = msalApp.getActiveAccount();

  if (!account) return;

  try {

    const resp = await msalApp.acquireTokenSilent({ scopes: SCOPES, account });

    accessToken = resp.accessToken;

  } catch(e) {

    const resp = await msalApp.acquireTokenPopup({ scopes: SCOPES });

    accessToken = resp.accessToken;

  }

}


function signOut() {

  if (!confirm('Abmelden?')) return;

  msalApp.logoutPopup();

  document.getElementById('login-screen').style.display = 'flex';

  document.getElementById('app-screen').style.display = 'none';

}


// ── ONEDRIVE SYNC ────────────────────────────────────────────

async function loadFromOneDrive() {

  if (!accessToken) return;

  try {

    const res = await fetch(`https://graph.microsoft.com/v1.0/me/drive/root:/${FILE_NAME}:/content`, {

      headers: { Authorization: 'Bearer ' + accessToken }

    });

    if (res.ok) {

      const data = await res.json();

      if (data && data.areas) S = data;

      toast('Daten geladen ✓');

    }

  } catch(e) { /* Datei existiert noch nicht – kein Problem */ }

}


async function saveToOneDrive() {

  if (!accessToken) { save(); return; }

  save();

  isSyncing = true;

  document.getElementById('sync-btn').textContent = 'Synchronisiert…';

  document.getElementById('sync-btn').classList.add('syncing');

  try {

    await getToken();

    await fetch(`https://graph.microsoft.com/v1.0/me/drive/root:/${FILE_NAME}:/content`, {

      method: 'PUT',

      headers: { Authorization: 'Bearer ' + accessToken, 'Content-Type': 'application/json' },

      body: JSON.stringify(S)

    });

    toast('Gespeichert in OneDrive ✓');

  } catch(e) {

    toast('Offline – lokal gespeichert');

  }

  isSyncing = false;

  document.getElementById('sync-btn').textContent = 'Synchronisieren';

  document.getElementById('sync-btn').classList.remove('syncing');

}


async function syncNow() { await saveToOneDrive(); }

function save() { try { localStorage.setItem('aze_ms', JSON.stringify(S)); } catch(e) {} }

function loadLocal() { try { const s = localStorage.getItem('aze_ms'); if(s) S = JSON.parse(s); } catch(e) {} }

}

// ── BEREICHE ─────────────────────────────────────────────────

function initSwatches() {

  document.getElementById('swatches').innerHTML = COLORS.map(c =>

    `<div class="sw${c===selColor?' sel':''}" style="background:${c}" onclick="pickColor('${c}',this)"></div>`

  ).join('');

}

function pickColor(c, el) {

  selColor = c;

  document.querySelectorAll('.sw').forEach(s => s.classList.remove('sel'));

  el.classList.add('sel');

}

function addArea() {

  const n = document.getElementById('b-name').value.trim();

  if(!n) return;

  const t = parseFloat(document.getElementById('b-target').value) || 0;

  S.areas.push({id:'a'+Date.now(), name:n, color:selColor, target:t});

  document.getElementById('b-name').value = '';

  document.getElementById('b-target').value = '';

  saveToOneDrive(); renderAreas(); updateSelects();

}

function delArea(id) {

  if(!confirm('Bereich löschen?')) return;

  S.areas = S.areas.filter(a => a.id !== id);

  saveToOneDrive(); renderAreas(); updateSelects();

}

function renderAreas() {

  initSwatches();

  const el = document.getElementById('area-list');

  if(!S.areas.length) { el.innerHTML = '<div class="empty">Noch keine Bereiche.</div>'; return; }

  el.innerHTML = S.areas.map(a => {

    const h = S.entries.filter(e => e.areaId===a.id).reduce((s,e)=>s+e.hours,0);

    return `<div class="card" style="padding:12px 16px;margin-bottom:8px">

      <div style="display:flex;align-items:center;gap:10px">

        <div style="width:10px;height:10px;border-radius:50%;background:${a.color};flex-shrink:0"></div>

        <div style="flex:1">

          <div style="font-size:14px">${a.name}</div>

          <div style="font-size:12px;color:var(--muted)">${h.toFixed(1)} h gesamt${a.target?` · Ziel ${a.target} h/Wo`:''}</div>

        </div>

        <button class="del-btn" onclick="delArea('${a.id}')">×</button>

      </div>

    </div>`;

  }).join('');

}

function updateSelects() {

  const opts = S.areas.map(a => `<option value="${a.id}">${a.name}</option>`).join('');

  document.getElementById('e-area').innerHTML = opts;

  const ra = document.getElementById('r-area');

  if(ra) ra.innerHTML = '<option value="">Alle</option>' + opts;

}


// ── EINTRÄGE ─────────────────────────────────────────────────

function addEntry() {

  const date = document.getElementById('e-date').value || today();

  const areaId = document.getElementById('e-area').value;

  const hours = parseFloat(document.getElementById('e-hours').value);

  const desc = document.getElementById('e-desc').value.trim();

  if(!areaId || !hours) return;

  S.entries.push({id:'e'+Date.now(), date, areaId, hours, desc});

  document.getElementById('e-hours').value = '1';

  document.getElementById('e-desc').value = '';

  saveToOneDrive(); renderToday();

}

function delEntry(id) {

  S.entries = S.entries.filter(e => e.id !== id);

  saveToOneDrive(); renderToday();

}

function renderToday() {

  const el = document.getElementById('today-entries');

  const t = today();

  const es = S.entries.filter(e => e.date===t).sort((a,b)=>b.id.localeCompare(a.id));

  if(!es.length) { el.innerHTML=''; return; }

  const tot = es.reduce((s,e)=>s+e.hours,0);

  el.innerHTML = `<div class="card"><div style="display:flex;justify-content:space-between;align-items:center;margin-bottom:10px">

    <div class="card-title" style="margin:0">Heute</div>

    <div style="font-size:14px;font-weight:600">${tot.toFixed(1)} h</div></div>` +

    es.map(e => {

      const a = area(e.areaId); if(!a) return '';

      return `<div class="entry">

        <div class="entry-dot" style="background:${a.color}"></div>

        <div class="entry-body">

          <div class="entry-name">${a.name}</div>

          ${e.desc ? `<div class="entry-sub">${e.desc}</div>` : ''}

        </div>

        <div class="entry-h">${e.hours.toFixed(2)} h</div>

        <button class="del-btn" onclick="delEntry('${e.id}')">×</button>

      </div>`;

function getWeekDates(off) {

  const d = new Date(); d.setDate(d.getDate() - d.getDay() + 1 + off*7);

  return Array.from({length:7},(_,i)=>{ const x=new Date(d); x.setDate(d.getDate()+i); return x.toISOString().slice(0,10); });

}

function exportCSV() {

  const rows = [['Datum','Bereich','Stunden','Beschreibung']];

  [...S.entries].sort((a,b)=>a.date.localeCompare(b.date)).forEach(e => {

    const a = area(e.areaId);

    rows.push([e.date, a?a.name:'', e.hours.toFixed(2), e.desc||'']);

  });

  const csv = rows.map(r=>r.map(c=>'"'+String(c).replace(/"/g,'""')+'"').join(',')).join('\n');

  const blob = new Blob(['\uFEFF'+csv],{type:'text/csv;charset=utf-8;'});

  const url = URL.createObjectURL(blob);

  const a = document.createElement('a'); a.href=url; a.download='arbeitszeit.csv'; a.click();

  URL.revokeObjectURL(url);

}
